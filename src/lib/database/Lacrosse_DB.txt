CREATE TABLE Game_ID
(
GAMEID INT NOT NULL,
primary key (GAMEID));


CREATE TABLE Sheet_ID
(
SHEET_ID INT NOT NULL,
Primary key (SHEET_ID));


CREATE TABLE Team_Name
(
TEAM_NAME varchar(100),
Primary key (TEAM_NAME));


CREATE TABLE PLAYERS(
PLAYER_NAME varchar(100),
PLAYER_NUMBER INT NOT NULL,
TEAM_NAME varchar(100),
POSITION varchar(100),
QUARTERS INT,
ATTEMPTED_SHOTS INT,
GOALS INT,
FAILED_SHOTS INT,
GROUND_BALLS INT,
primary key(PLAYER_NAME, PLAYER_NUMBER),
foreign key(TEAM_NAME) references Team_Name ON DELETE CASCADE ON UPDATE CASCADE);


CREATE TABLE Game_Stats(
SHEET_ID INT,
SIDE varchar(100),
QUARTER INT,
SHOTS INT,
CLEARS_PASS INT,
CLEARS_FAIL INT,
EXTRA_MAN_SCORE INT,
EXTRA_MAN_FAIL INT,
FACEOFF_WIN INT,
FACEOFF_LOSS INT,
primary key (SHEET_ID, SIDE));


CREATE TABLE Individual_Score(
SHEET_ID INT,
SIDE varchar(100),
PLAYER_NUMBER INT NOT NULL,
PLAYER_NAME varchar(100),
GOALS INT,
ATTEMPTS INT,
FAILS INT,
foreign key (SHEET_ID) references Game_Stats(SHEET_ID) ON DELETE CASCADE ON UPDATE CASCADE);


CREATE TABLE Timeouts(
SHEET_ID INT,
SIDE varchar(100),
HALF_1_TIME INT,
HALF_2_TIME INT,
OT_1_TIME INT,
OT_2_TIME INT,
foreign key (SHEET_ID) references Game_Stats(SHEET_ID) ON DELETE CASCADE ON UPDATE CASCADE);


CREATE TABLE Sheets(
GAME_ID INT NOT NULL,
SHEET_ID INT NOT NULL,
primary key (SHEET_ID),
foreign key (GAME_ID) references Game_ID(GAME_ID) ON DELETE CASCADE ON UPDATE CASCADE);


CREATE TABLE Penalties_Fouls(
SHEET_ID INT NOT NULL,
SIDE varchar(100),
PT INT NOT NULL,
PLAYER_NUMBER INT NOT NULL,
INFRACTION varchar(100),
TIME INT NOT NULL,
Foreign key (SHEET_ID) references Sheet_ID(SHEET_ID) ON DELETE CASCADE ON UPDATE CASCADE);


CREATE TABLE Saves(
SHEET_ID INT NOT NULL,
SIDE varchar(100),
PLAYER_NAME varchar(100),
PLAYER_NUMBER INT NOT NULL,
QUARTER_1 INT,
QUARTER_2 INT,
QUARTER_3 INT,
QUARTER_4 INT,
OT INT,
TOTAL INT,
foreign key (SHEET_ID) references Sheet_ID(SHEET_ID) ON DELETE CASCADE ON UPDATE CASCADE);


CREATE TABLE Sheet_Info(
SHEET_ID INT NOT NULL,
DATE INT NOT NULL,
SITE varchar(100),
START_TIME INT NOT NULL,
SCOREKEEPER varchar(100),
OPPONENT_SCORE INT,
TIMEKEEPER VARCHAR(100),
HEAD_OFFICAL varchar(100),
UMPIRE varchar(100),
FIELD_JUDGE varchar(100),
foreign key (SHEET_ID) references Sheet_ID(SHEET_ID) ON DELETE CASCADE ON UPDATE CASCADE);
